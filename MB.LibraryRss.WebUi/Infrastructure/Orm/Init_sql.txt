SET NUMERIC_ROUNDABORT OFF
GO
SET ANSI_PADDING, ANSI_WARNINGS, CONCAT_NULL_YIELDS_NULL, ARITHABORT, QUOTED_IDENTIFIER, ANSI_NULLS ON
GO
IF EXISTS (SELECT * FROM tempdb..sysobjects WHERE id=OBJECT_ID('tempdb..#tmpErrors')) DROP TABLE #tmpErrors
GO
CREATE TABLE #tmpErrors (Error int)
GO
SET XACT_ABORT ON
GO
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE
GO
BEGIN TRANSACTION
GO
PRINT N'Creating [dbo].[Element]'
GO
CREATE TABLE [dbo].[Element]
(
[Id] [int] NOT NULL IDENTITY(1, 1),
[Data] [text] COLLATE Latin1_General_CI_AS NOT NULL,
[Inserted] [datetime] NOT NULL CONSTRAINT [DF_ChildElement_Inserted] DEFAULT (getdate())
)
GO
IF @@ERROR<>0 AND @@TRANCOUNT>0 ROLLBACK TRANSACTION
GO
IF @@TRANCOUNT=0 BEGIN INSERT INTO #tmpErrors (Error) SELECT 1 BEGIN TRANSACTION END
GO
PRINT N'Creating primary key [PK_ChildElement] on [dbo].[Element]'
GO
ALTER TABLE [dbo].[Element] ADD CONSTRAINT [PK_ChildElement] PRIMARY KEY CLUSTERED  ([Id])
GO
IF @@ERROR<>0 AND @@TRANCOUNT>0 ROLLBACK TRANSACTION
GO
IF @@TRANCOUNT=0 BEGIN INSERT INTO #tmpErrors (Error) SELECT 1 BEGIN TRANSACTION END
GO
PRINT N'Creating [dbo].[InsertElement]'
GO
CREATE PROCEDURE [dbo].[InsertElement]
  (
    @Data TEXT
  )
AS
  BEGIN
    SET NOCOUNT ON;

    INSERT  INTO [Element]
            ( [Data] )
    VALUES
            ( @Data )

  END
GO
IF @@ERROR<>0 AND @@TRANCOUNT>0 ROLLBACK TRANSACTION
GO
IF @@TRANCOUNT=0 BEGIN INSERT INTO #tmpErrors (Error) SELECT 1 BEGIN TRANSACTION END
GO
PRINT N'Creating [dbo].[DeleteElements]'
GO
CREATE PROCEDURE [dbo].[DeleteElements]
AS
  BEGIN
    SET NOCOUNT ON;

    DELETE FROM
      [Element]

  END
GO
IF @@ERROR<>0 AND @@TRANCOUNT>0 ROLLBACK TRANSACTION
GO
IF @@TRANCOUNT=0 BEGIN INSERT INTO #tmpErrors (Error) SELECT 1 BEGIN TRANSACTION END
GO
PRINT N'Creating [dbo].[ReadElements]'
GO
CREATE PROCEDURE [dbo].[ReadElements]
AS
  BEGIN
    SET NOCOUNT ON;

    SELECT Data, Inserted FROM
      [Element]

  END
GO
IF @@ERROR<>0 AND @@TRANCOUNT>0 ROLLBACK TRANSACTION
GO
IF @@TRANCOUNT=0 BEGIN INSERT INTO #tmpErrors (Error) SELECT 1 BEGIN TRANSACTION END
GO
PRINT N'Altering permissions on [dbo].[InsertElement]'
GO
GRANT EXECUTE ON  [dbo].[InsertElement] TO [public]
GO
PRINT N'Altering permissions on [dbo].[ReadElements]'
GO
GRANT EXECUTE ON  [dbo].[ReadElements] TO [public]
GO
PRINT N'Altering permissions on [dbo].[DeleteElements]'
GO
GRANT EXECUTE ON  [dbo].[DeleteElements] TO [public]
GO
IF EXISTS (SELECT * FROM #tmpErrors) ROLLBACK TRANSACTION
GO
IF @@TRANCOUNT>0 BEGIN
PRINT 'The database update succeeded'
COMMIT TRANSACTION
END
ELSE PRINT 'The database update failed'
GO
DROP TABLE #tmpErrors
GO
